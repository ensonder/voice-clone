# ===== CELL 1: Install ===== !pip install -U qwen-tts soundfile huggingface_hub transformers tokenizers flask pyngrok # ===== CELL 2: Download Models ===== from huggingface_hub import snapshot_download snapshot_download(repo_id="Qwen/Qwen3-TTS-Tokenizer-12Hz", local_dir="./Qwen3-TTS-Tokenizer-12Hz") snapshot_download(repo_id="Qwen/Qwen3-TTS-12Hz-1.7B-CustomVoice", local_dir="./Qwen3-TTS-12Hz-1.7B-CustomVoice") snapshot_download(repo_id="Qwen/Qwen3-TTS-12Hz-1.7B-Base", local_dir="./Qwen3-TTS-12Hz-1.7B-Base") # ===== CELL 3: Start Server ===== import torch, base64, io, threading from flask import Flask, request, jsonify from flask_cors import CORS from pyngrok import ngrok import soundfile as sf from qwen_tts import Qwen3TTSModel import numpy as np app = Flask(__name__) CORS(app) print("Loading CustomVoice model...") custom_model = Qwen3TTSModel.from_pretrained( "./Qwen3-TTS-12Hz-1.7B-CustomVoice", device_map="cuda:0", dtype=torch.bfloat16) print("Loading Base (clone) model...") clone_model = Qwen3TTSModel.from_pretrained( "./Qwen3-TTS-12Hz-1.7B-Base", device_map="cuda:0", dtype=torch.bfloat16) def wav_to_base64(wav, sr): buf = io.BytesIO() sf.write(buf, wav, sr, format='WAV') buf.seek(0) return base64.b64encode(buf.read()).decode() @app.route('/health', methods=['GET']) def health(): return jsonify({"status": "ok", "message": "VoiceCraft is ready!"}) @app.route('/custom', methods=['POST']) def custom(): d = request.json wavs, sr = custom_model.generate_custom_voice( text=d.get('text','Hello world'), language=d.get('language','English'), speaker=d.get('speaker','Ryan'), instruct=d.get('instruct','') or None) return jsonify({"audio": wav_to_base64(wavs[0], sr), "sr": sr}) @app.route('/clone', methods=['POST']) def clone(): d = request.json audio_bytes = base64.b64decode(d['ref_audio']) buf = io.BytesIO(audio_bytes) import soundfile as sf2 wav_data, wav_sr = sf2.read(buf) wavs, sr = clone_model.generate_voice_clone( text=d.get('text','Hello world'), language=d.get('language','English'), ref_audio=(wav_data, wav_sr), ref_text=d.get('ref_text','')) return jsonify({"audio": wav_to_base64(wavs[0], sr), "sr": sr}) @app.route('/design', methods=['POST']) def design(): d = request.json wavs, sr = custom_model.generate_custom_voice( text=d.get('text','Hello world'), language=d.get('language','English'), speaker=d.get('speaker','Ryan'), instruct=d.get('instruct','')) return jsonify({"audio": wav_to_base64(wavs[0], sr), "sr": sr}) ngrok.set_auth_token("YOUR_NGROK_TOKEN_HERE") # get free token at ngrok.com tunnel = ngrok.connect(5000) print(f"\n🎉 YOUR URL: {tunnel.public_url}\n") print("Paste this URL into VoiceCraft Studio!") threading.Thread(target=lambda: app.run(port=5000)).start()
